```init.tpl

    #!/bin/bash 
    echo "app1-ip = ${public_ip}" >> etc/ip.config

```

```templates.tf
data "template_file" "ip-tpl" {
        template = "${file("init.tpl")}"
        vars {
            public_ip = "${aws_instance.app1.public_ip}"
        }
    }
```

```instance.tf
resource "aws_instance" "app2" { 
        user_data = "${data.template_file.ip-tpl.rendered}"
    }
```
